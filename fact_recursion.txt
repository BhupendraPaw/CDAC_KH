public class trial {
    static int fact(int n){
        if (n==0)
        return 1;
    else
    return n*fact(n-1);
    }

    public static void main(String[] args) {
        int i,f=1;
     Scanner a =new Scanner(System.in);
     System.out.println("Enter a number :");
     int x = a.nextInt();

    }
